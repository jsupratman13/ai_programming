--compile tex document--
pdflatex [filename].tex

--open pdf document--
open [filename].pdf
(or if the above doesnt work)
explorer [filename].pdf

--run python--
python [filename]

--run python3--
python -3 [filename]	(#!python3 on the script)

--complie c files--
cc [filename].c -o [filename]
(or)
gcc [filename].c -o [filename]
(or)
make [filename] (assuming that [filename.c] exist)

--compile c++ files--
g++ [filename].cpp -o [filename]
(or)
make [filename] (assuming that [filename.c] exist)


--complie c++/c files with opencv--
gcc [filename] pkg-config --cflags opencv pkg-config --libs opencv

--complie c++/c files with windows.h--
g++ -mwindows [filename]

--compile c+/c files with opengl--
g++ (gcc) [filename] -lfreeglut -lglu32 -lopengl32

--call c from python--
gcc -Wall -fPIC -c [filename].c -I"C:\python27\include"
gcc -shared -o [filename].so [filename].o
(in python import ctypes then [name] = ctypes.CDLL("./[filename].so")

--compile c/c+ files with eigen--
g++ [filename] -I"C:\mingw\eigen"


--run c/c++ files--
(after compile)
./[filename] (assuming that output file exit)
(or)
./a.exe (default output file)


--make c files--
make 		(search for Makefile)
make -f [filename]

--compile c/cpp but object file--
gcc -c [filename]

--symbolic information/warning message--
(c)
cc -g -Wall [filename].c -o executable
(cpp)
g++ -g -Wall [filename].c -o executable

--compile c that uses math function--
cc [filename].c -o executable -lm

--compile cpp with xlib graphic function--
g++ [filename].cpp -o executable -lX11

--compile multiple source code--
(c)
cc [filename].c [filename2].c (...) -o executable
(cpp)
g++ [filename].cpp [filename2].cpp (...) -o [filename3]



--pip install--
easy_install.pip

--install .whl files--
pip install [filename]

--update .whl files--
pip install --upgrade [filename]


-activate gnuplot--
gnuplot


--kill process in simulation--
kill -9 %1
kill -9 [number from ps aux]
(or)
sudo killall -KILL python
(for window)
taskkill /f /im [number from tasklist]


--check process--
ps aux | grep [file name]
(for window)
tasklist /v

--create new file--
touch [filename]
