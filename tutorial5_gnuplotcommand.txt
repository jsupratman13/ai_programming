*easily plot graph and data
*2d,3d graph, plot from data using outside file such as excel

--plot graph--
plot [function]
splot [function] 		(for 3d graph)
plot [function] title '[name]'  (line name for function)
plot [function], [function] 	(for multiple function)


--plot last function--
replot

--set variable--
f(x) = [function1]
g(x,y) = [function2] 		(use any function)

--reset all option-
reset

--exit gnuplot--
exit
(or)
quit

--edit graph--
set xrange [-10:10] 		(set range from -10 to 10 in x axis)
(or) set xr [-10:10]
set yrange [-10:10] 		(set range from -10 to 10 in y axis)
set grid 			(set grid)
set size ratio [value]  	(set graph size on window)
set title '[name]'		(set title)
set xlabel '[name]'		(set label in x axis)
set ylabel '[name]'		(set label in y axis)
set autoscale			(auto scale)
set logscale (x,y)		(set log data)
set xtics auto			(auto tics)
set ytics (1,2,3)		(set ytics on 1,2,3)

--remove modification from graph--
unset [type]			(label, tics, logscale etc.)

--multiplot--
set multiplot 			(set multiple plot)
unset multiplot			(unset multiple plot

--save graph as output (save in current dircetory check using pwd)--
set terminal [type] 		(latex, jpg, png etc)
set output '[name].[type]'	
replot				(basically edit file on spot)


--curve fit with data--
*must have a plotting data
f1(x) = a*x + b			(pick a function plus variable to fit data)
a = 10; b=1			(pick initial value)
fit f(x) '[name].dat' u 1:2 via a,b
				(fit parameter to data)

--gnuplot script--
*must be name [filename].p
*the # are treated as comment
load '[filename].p'

--call data from outside file--
*must be name [filename].dat
*data must be arrange in columns, seperated by tabs or spaces
*the # are treated as comment
plot "[filename].dat" using 1:2	(use 2nd row in data)
plot "[name].dat" t 'wak',\	(u= using, t = tilte, w= with)
     "[name].dat" w lines	(lines, linespoint)

sqrt(x)				(square root)

x**n				(x^n)

log(x), log10(x)		(log)
cos(x), sin(x), tan(x)		(trignometry)

rand(x)  			(random number)
